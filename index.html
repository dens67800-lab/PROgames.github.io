    // === КОММЕНТАРИИ ===

    let comments = [];

   

    function loadComments() {

        const saved = localStorage.getItem('progames_comments');

        if(saved) comments = JSON.parse(saved);

        else comments = [

            { id: Date.now(), author: "PROgamer", text: "Отличные игры! Всё работает!", date: new Date().toLocaleString() },

            { id: Date.now()+1, author: "GameMaster", text: "Идея: добавить арканоид", date: new Date().toLocaleString() }

        ];

        renderComments();

    }

   

    function saveComments() {

        localStorage.setItem('progames_comments', JSON.stringify(comments));

        renderComments();

    }

   

    function renderComments() {

        const container = document.getElementById('commentsList');

        if(!container) return;

        container.innerHTML = '';

        for(let c of comments) {

            const div = document.createElement('div');

            div.className = 'comment-item';

            div.innerHTML = '<div class="comment-author">' + escapeHtml(c.author) + '</div><div class="comment-text">' + escapeHtml(c.text) + '</div><div class="comment-date">📅 ' + c.date + '</div><button class="delete-comment" data-id="' + c.id + '">🗑️ удалить</button>';

            container.appendChild(div);

        }

        document.querySelectorAll('.delete-comment').forEach(btn => {

            btn.addEventListener('click', function() {

                comments = comments.filter(c => c.id !== parseInt(this.getAttribute('data-id')));

                saveComments();

            });

        });

    }

   

    function escapeHtml(str) {

        return str.replace(/[&<>]/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;'}[m] || m));

    }

   

    function addComment(author, text) {

        if(!author.trim()) author = 'Аноним';

        if(!text.trim()) return;

        comments.unshift({ id: Date.now(), author: author.substring(0,20), text: text.substring(0,200), date: new Date().toLocaleString() });

        saveComments();

    }

   

    document.getElementById('submitCommentBtn').addEventListener('click', function() {

        addComment(document.getElementById('authorName').value, document.getElementById('commentText').value);

        document.getElementById('authorName').value = '';

        document.getElementById('commentText').value = '';

    });

   

    // === КНОПКИ ===

    const modal = document.getElementById('gameModal');

    const frameContainer = document.getElementById('gameFrameContainer');

    const closeBtn = document.getElementById('closeModalBtn');

   

    function openGame(code) {

        const iframe = document.createElement('iframe');

        iframe.src = 'data:text/html;charset=utf-8,' + encodeURIComponent(code);

        iframe.style.width = '100%';

        iframe.style.height = '550px';

        iframe.style.border = 'none';

        iframe.style.borderRadius = '16px';

        frameContainer.innerHTML = '';

        frameContainer.appendChild(iframe);

        modal.style.display = 'flex';

        document.body.style.overflow = 'hidden';

    }

   

    function closeGame() {

        modal.style.display = 'none';

        frameContainer.innerHTML = '';

        document.body.style.overflow = '';

    }

   

    function downloadGame(code, filename) {

        const blob = new Blob([code], {type: 'text/html'});

        const a = document.createElement('a');

        a.href = URL.createObjectURL(blob);

        a.download = filename;

        document.body.appendChild(a);

        a.click();

        document.body.removeChild(a);

        URL.revokeObjectURL(blob);

    }

   

    document.querySelectorAll('.play-btn').forEach(btn => {

        btn.addEventListener('click', function() {

            const game = this.getAttribute('data-game');

            if(game === 'snake') openGame(snakeGameCode);

            if(game === 'tetris') openGame(tetrisGameCode);

            if(game === 'time') openGame(timeGameCode);

            if(game === 'onebyte') openGame(oneByteGameCode);

        });

    });

   

    document.querySelectorAll('.download-btn').forEach(btn => {

        btn.addEventListener('click', function() {

            const game = this.getAttribute('data-game');

            if(game === 'snake') downloadGame(snakeGameCode, 'snake_game.html');

            if(game === 'tetris') downloadGame(tetrisGameCode, 'tetris_game.html');

            if(game === 'time') downloadGame(timeGameCode, 'time_reaction.html');

            if(game === 'onebyte') downloadGame(oneByteGameCode, 'one_byte_game.html');

        });

    });

   

    closeBtn.addEventListener('click', closeGame);

    modal.addEventListener('click', function(e) {

        if(e.target === modal) closeGame();

    });

   

    loadComments();
